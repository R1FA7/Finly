import Menu from './Menu';
import { useState } from 'react';

const Category = ({ data, index, activeIndex, setActiveIndex }) => {
  return (
    <div className="w-2/3 mx-auto border border-2 rounded-md shadow-lg m-2 p-4 bg-gray-200">
      <div
        className="flex justify-between items-center font-bold cursor-pointer"
        onClick={() => setActiveIndex(activeIndex === index ? null : index)}
      >
        <p>
          {data.menu} ({data.items.length})
        </p>
        <button className="font-bold">
          {activeIndex === index ? '▲' : '▼'}
        </button>
      </div>

      {activeIndex === index && <Menu data={data.items} />}
    </div>
  );
};
export default Category;

}
